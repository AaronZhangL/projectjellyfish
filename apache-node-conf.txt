# You should never run a node app as root (port 80 or 443)
# so let's run this through a reverse proxy with apache (will make setting
# up SSL so much easier) too.
#
# Also, on RHEL/CentOS (with SELinux enabled) run this:
# /usr/sbin/setsebool httpd_can_network_connect 1

<VirtualHost *:80>
    ServerAdmin noreply@jellyfish-ux-test.dpi.bah.com
    ServerName jellyfish-ux-test.dpi.bah.com

    ProxyRequests off

    <Proxy *>
        Order deny,allow
        Allow from all
    </Proxy>

    <Location />
        ProxyPass http://localhost:5000/
        ProxyPassReverse http://localhost:5000/
    </Location>
</VirtualHost>
